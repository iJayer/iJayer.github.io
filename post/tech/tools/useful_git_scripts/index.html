#!/bin/sh

set -e

if [ 0 -eq $# ]; then
        echo "Usage: gcp <commit>"
        exit 1
fi

git add -A
git commit -m "$1"
git push
git status

#!/bin/sh

set -e

if [ 0 -eq $# ]; then
        echo "Usage: gtd <tag>"
        exit 1
fi

git tag -d $1
git push origin :refs/tags/$1
git tag -n
